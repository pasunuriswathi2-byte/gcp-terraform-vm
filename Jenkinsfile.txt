pipeline {
  agent any

  environment {
    // path inside agent where we'll write the service account file
    GCP_SA_JSON = 'gcp-sa.json'
  }

  stages {
    stage('Checkout') {
      steps {
        checkout(scm)
      }
    }

    stage('Prepare credentials') {
      steps {
        // using credentials binding plugin to get the secret file content
        withCredentials([file(credentialsId: 'gcp-sa-key', variable: 'GCP_SA_FILE')]) {
          sh '''
            # copy Jenkins-managed secret file to known path
            cp "$GCP_SA_FILE" ${GCP_SA_JSON}
            chmod 600 ${GCP_SA_JSON}
            ls -l ${GCP_SA_JSON}
            cat ${GCP_SA_JSON} | head -n 2
          '''
        }
      }
    }

    stage('Terraform Init') {
      steps {
        sh '''
          # Install terraform if not available (for demo) - ideally install on agent image
          if ! command -v terraform >/dev/null 2>&1; then
            echo "Terraform not found. Please install Terraform on the agent."
            exit 1
          fi

          export GOOGLE_APPLICATION_CREDENTIALS=${GCP_SA_JSON}
          terraform init -input=false
        '''
      }
    }

    stage('Terraform Plan') {
      steps {
        sh '''
          export GOOGLE_APPLICATION_CREDENTIALS=${GCP_SA_JSON}
          terraform plan -input=false -out=tfplan
        '''
      }
    }

    stage('Terraform Apply') {
      steps {
        input message: "Apply terraform changes?", ok: "Apply"
        sh '''
          export GOOGLE_APPLICATION_CREDENTIALS=${GCP_SA_JSON}
          terraform apply -input=false tfplan
        '''
      }
    }
  }

  post {
    always {
      sh 'rm -f ${GCP_SA_JSON} tfplan'
    }
  }
}
